A macro to easily define wrappers for foreign types.

Examples

#[macro_use]
extern crate foreign_types;

foreign_type! {
    type CType = openssl_sys::SSL;
    fn drop = openssl_sys::SSL_free;
    /// Documentation for the owned type.
    pub struct Ssl;
    /// Documentation for the borrowed type.
    pub struct SslRef;
}
